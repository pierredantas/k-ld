\newcommand{\bars}[1]{%
  \begin{tikzpicture}[baseline=-0.4ex]
    \foreach \i in {1,...,5}{%
      \pgfmathparse{\i <= #1 ? 1 : 0}%
      \ifnum\pgfmathresult=1
        \fill[black]     (\i*0.12-0.12, 0) rectangle (\i*0.12-0.03, 0.22);
      \else
        \fill[black!18]  (\i*0.12-0.12, 0) rectangle (\i*0.12-0.03, 0.22);
      \fi
    }
  \end{tikzpicture}%
}
